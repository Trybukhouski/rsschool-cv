Cata: Run-length encoding 

              
function runLengthEncoding(str){
  const result = [];
  let counter = 1;
  for (let i = 0; i < str.length; i++) {
    if(str[i] === str[i + 1]) {
      counter++;
    } else {
      result.push([counter, str[i]]);
      counter = 1;
    }
  }
  return result;
}

Cata: Sorting by bits 

              
function sortByBit(arr) {
  return arr.sort(function(a, b) {
    if(bitCounter(a) !== bitCounter(b)) {
      return bitCounter(a) - bitCounter(b);
    } else {
        return a - b;
      }
  })
}

function bitCounter(num) {
  return num
    .toString(2)
    .split('')
    .filter(el => el === '1')
    .length
}

Cata: Array Deep Count 

              
function deepCount(a) {
  let count = 0;
  function recursion(a) {
    a.forEach(el => {
      count++;
      if(Array.isArray(el)) recursion(el);
    })
    return count;
  }
  return recursion(a);
}
